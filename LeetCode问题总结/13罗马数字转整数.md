1、 在进行 if 判断时，不要先判断不等于，要先判断等于，否则会出错

```JavaScript
if (s[i + 1] == 'V') {
  romanToInt += 4
  i++
} else if (s[i + 1] == 'X') {
  romanToInt += 9
  i++
} else {
  romanToInt += 1
}
```

2、 代码超时

```JavaScript
/**
 * @param {string} s
 * @return {number}
 */
var romanToInt = function(s) {
    while(s.length>=1&&s.length<=15){
        romanToInt=0;
        for(var i=0;i<s.length;i++){
            switch(s[i]){
                case 'I':{
                    if(s[i+1]=='V'){
                        romanToInt+=4;
                        i++;
                    }else if(s[i+1]=='X'){
                            romanToInt+=9
                            i++;
                        }else{
                            romanToInt+=1
                        }
                        break;
                    }
                case 'V':{
                    romanToInt+=5;
                    break;
                }
                case 'X':{
                    if(s[i+1]=='L'){
                        romanToInt+=40
                        i++
                    }else if(s[i+1]=='C'){
                        romanToInt+=90
                        i++
                    }else{
                        romanToInt+=10
                    }
                    break
                }
                case 'L':{
                    romanToInt+=50
                    break
                }
                case 'C':{
                    if(s[i+1]=='D'){
                        romanToInt+=400
                        i++
                    }else if(s[i+1]=='M'){
                        romanToInt+=900
                        i++
                    }else{
                        romanToInt+=100
                    }
                    break
                }
                case 'D':{
                    romanToInt+=500
                    break
                }
                case 'M':{
                    romanToInt+=1000
                    break
                }
                default:{
                    break
                }
            }
        }
    }
    return false
};
```

3、没必要加while判断`1 <= s.length <= 15`
